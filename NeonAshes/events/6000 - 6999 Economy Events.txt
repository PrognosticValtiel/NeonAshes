### 6000-6009 - Economic Expansion and Contraction ###

country_event = { # Economic Depression to Recession
	id = 6000
	title = "EVT_6000_NAME"
	desc = "EVT_6000_DESC"
	picture = "economics_positive_pic"
	allow_multiple_instances = yes
	trigger = { 
		has_country_modifier = economic_depression		# We are in a depression
		OR = {							# We don't have hyperinflation or hyperdeflation
			has_country_modifier = rapid_deflation
			has_country_modifier = deflation
			has_country_modifier = very_low_inflation
			has_country_modifier = low_inflation
			has_country_modifier = moderate_inflation
			has_country_modifier = high_inflation
			has_country_modifier = very_high_inflation
			has_country_modifier = extremely_high_inflation
		}
		NOT = { has_country_modifier = contractionary_fiscal_policy }	# Contractionary fiscal policy pushes down
	}
	mean_time_to_happen = {
		months = 12
		modifier = {						# More moderate inflation means faster recovery
			factor = 0.5
			OR = {
				has_country_modifier = deflation
				has_country_modifier = very_low_inflation
				has_country_modifier = low_inflation
				has_country_modifier = moderate_inflation
				has_country_modifier = high_inflation
				has_country_modifier = very_high_inflation
			}
		}
		modifier = {
			factor = 0.5
			has_country_modifier = expansionary_fiscal_policy	# Expansionary fiscal policy pushes up
		}
		modifier = {
			factor = 1.2
			has_country_modifier = semi_centralized_communities	# Hard to grow with low interconnectedness
		}
		modifier = {
			factor = 1.4
			has_country_modifier = decentralized_communities	# Hard to grow with low interconnectedness
		}
		modifier = {
			factor = 1.6
			has_country_modifier = frontier_communities		# Hard to grow with low interconnectedness
		}
		modifier = {
			factor = 1.8
			has_country_modifier = autarkic_communities		# Hard to grow with low interconnectedness
		}
		modifier = {
			factor = 2
			has_country_modifier = survivalist_communities		# Hard to grow with low interconnectedness
		}
	}
	option = {
		name = "EVT_6000_OPTA"
		remove_country_modifier = economic_depression
		remove_country_modifier = economic_recession		# Sanity check
		remove_country_modifier = economic_stagnation
		remove_country_modifier = economic_expansion		# Just in case they got doubled up
		remove_country_modifier = economic_boom
		add_country_modifier = {
			name = economic_recession
			duration = -1
		}
	}
}

		has_country_modifier = hyperdeflation			# We have hyperdeflation

			has_country_modifier = economic_stagnation
			has_country_modifier = economic_expansion
			has_country_modifier = economic_boom
		}
		NOT = { has_country_modifier = contractionary_monetary_policy } # Contractionary monetary policy halts inflation growth
	}
	mean_time_to_happen = {
		months = 12
		modifier = {						# Economic expansion makes it go fast
			factor = 0.5
			has_country_modifier = economic_expansion

		modifier = {						# Bigger expansion makes it faster
			factor = 0.2
			has_country_modifier = economic_boom
		}
		modifier = {
			factor = 0.5
			has_country_modifier = expansionary_monetary_policy	# Expansionary monetary policy boosts inflation growth
		}
	}
	option = {
		name = "EVT_6010_OPTA"
		remove_country_modifier = hyperdeflation
		remove_country_modifier = rapid_deflation
		remove_country_modifier = deflation
		remove_country_modifier = very_low_inflation		# Sanity check
		remove_country_modifier = low_inflation
		remove_country_modifier = moderate_inflation
		remove_country_modifier = high_inflation		# Just in case they got doubled up
		remove_country_modifier = very_high_inflation
		remove_country_modifier = extremely_high_inflation
		remove_country_modifier = hyperinflation

			duration = -1
		}

country_event = { # Decreasing Interconnectedness (to Semi-Centralized Communities)
	id = 6040
	title = "EVT_6040_NAME"
	desc = "EVT_6040_DESC"
	picture = "interconnectedness_semi_centralized_pic"
	allow_multiple_instances = yes
	trigger = {
		NOT = {						# Currently fully centralized country
			OR = {
				has_country_modifier = semi_centralized_communities
				has_country_modifier = decentralized_communities
				has_country_modifier = frontier_communities
				has_country_modifier = autarkic_communities
				has_country_modifier = survivalist_communities
			}
		}
		OR = {						# It can be triggered by either...
			national_provinces_occupied = 0.15	# Enemy occupation of at least 15 percent or more of the country
			AND = {					# Or...
				num_of_revolts = 1		# At least one province is occupied by rebels
				average_militancy = 5		# Moderate militancy
			}
		}
		NOT = { has_country_modifier = decreasing_interconnectedness }	# Brief cooldown period
	}
	mean_time_to_happen = {
		months = 3
		modifier = {		# A depression hastens it a lot
			factor = 0.5
			has_country_modifier = economic_depression
		}
		modifier = {		# A stagnant economy buys a bit of time
			factor = 2
			has_country_modifier = economic_stagnation
		}
		modifier = {		# A growing economy buys some time
			factor = 3
			has_country_modifier = economic_expansion
		}
		modifier = {		# A booming economy buys a lot of time
			factor = 4
			has_country_modifier = economic_boom
		}
	}
	option = {
		name = "EVT_6040_OPTA"
		add_country_modifier = {				# Semi-Centralized Communities
			name = semi_centralized_communities
			duration = -1
		}
		add_country_modifier = {				# Brief cooldown period
			name = decreasing_interconnectedness
			duration = 30
		}
		remove_country_modifier = increasing_interconnectedness	# Reset interconnectedness growth
		prestige = -2						# Ouch
	}
}
